\subsection*{Problem 1}
\begin{multicols}{2}
    \begin{lstlisting}
        for i = 1 to n
            j = 1
            while j<= i do
                j = 2.5 * j
    \end{lstlisting}
    \columnbreak
    \footnotesize
    \begin{tabular}{lcc}
        cost  & time                 \\
        $C_1$ & $n + 1$              \\
        $C_2$ & $n$                  \\
        $C_3$ & $(n + 1) \log_{2.5}$ \\
        $C_4$ & $n \log_{2.5} n$
    \end{tabular}
\end{multicols}

Detailed Runtime:

\begin{equation*}
    C_1 \cdot (n + 1) + C_2 \cdot n + C_3 \cdot \lceil (n+1) \log_{2.5} \rceil + C_4 \cdot \lceil n \log_{2.5}n \rceil
\end{equation*}

Asymptotic Runtime:
\begin{equation*}
    \Theta(n \log n)
\end{equation*}

\subsection*{Problem 2}

        def factorial(n):
            x = 1
            while n > 1:
                x = x * n
                n = n - 1
            return x
    \end{lstlisting}
    \columnbreak

    \begin{tabular}{lcc}
